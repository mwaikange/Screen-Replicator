Fix 1 — Map must only show verified incidents
The query must filter to only incidents where verification_level is greater than 0:
javascriptconst { data: incidents } = await supabase
  .from("incidents")
  .select("id, type_id, title, lat, lng, status, verification_level, incident_types(label, severity)")
  .gte("lat", southLat)
  .lte("lat", northLat)
  .gte("lng", westLng)
  .lte("lng", eastLng)
  .neq("status", "archived")
  .gt("verification_level", 0)        // ONLY verified incidents on map
  .limit(200)
