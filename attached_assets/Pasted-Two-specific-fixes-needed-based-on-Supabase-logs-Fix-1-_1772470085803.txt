javascriptimport ngeohash from 'ngeohash'

// Get GPS coordinates first
const location = await Location.getCurrentPositionAsync({})
const lat = location.coords.latitude
const lng = location.coords.longitude
const geohash = ngeohash.encode(lat, lng, 7)

const { data, error } = await supabase
  .from("incidents")
  .insert({
    type_id: selectedTypeId,        // required — must be a number from incident_types table
    title: titleValue,              // required
    description: descriptionValue || null,
    town: townValue || null,
    lat: lat,                       // required — from GPS
    lng: lng,                       // required — from GPS
    geohash: geohash,               // required — compute with ngeohash
    area_radius_m: 200,
    created_by: userId              // required — current user's UUID
  })
  .select()
  .single()

if (error) {
  console.log("Incident insert error:", error.message, error.details, error.hint)
  throw new Error(error.message)
}
